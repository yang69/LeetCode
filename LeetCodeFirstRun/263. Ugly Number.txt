public class Solution {
    public boolean isUgly(int num) {
        if(num == 0) { //防止0导致下面的while语句陷入死循环
            return false;
        }
        while(num%2 == 0) {
            num /= 2;
        }
        while(num%3 == 0) {
            num /= 3;
        }
        while(num%5 == 0) {
            num /= 5;
        }
        if(num == 1) {
            return true;
        }
        return false;
    }
}
